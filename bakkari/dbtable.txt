CREATE TABLE `memes` (
  `meme_id` int NOT NULL AUTO_INCREMENT,
  `name` text NOT NULL,
  `filename` text NOT NULL,
   `tag` text NOT NULL,
   PRIMARY KEY (meme_id)
) ENGINE=MyISAM DEFAULT CHARSET=utf8;
